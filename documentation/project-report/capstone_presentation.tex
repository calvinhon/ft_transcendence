% ============================================================================
\begin{frame}{Technical Decisions \& Trade-Offs}
    \begin{block}{Database Choice: SQLite}
        \begin{itemize}
            \item \textbf{Decision:} SQLite for single-node deployment
            \item \textbf{Rationale:} Simplicity, no separate DB server needed
            \item \textbf{Trade-off:} Not suitable for multi-node horizontal scaling
            \item \textbf{Mitigation:} Could migrate to PostgreSQL if needed
        \end{itemize}
    \end{block}
    
    \begin{block}{Frontend Framework: React}
        \begin{itemize}
            \item \textbf{Decision:} React with TypeScript for type safety
            \item \textbf{Rationale:} Large ecosystem, extensive component libraries
            \item \textbf{Trade-off:} Larger bundle size compared to Vue/Svelte
            \item \textbf{Optimization:} Code splitting, lazy loading, tree shaking
        \end{itemize}
    \end{block}
    
    \begin{block}{Blockchain: Hardhat with Local Testnet}
        \begin{itemize}
            \item \textbf{Decision:} Local blockchain for development
            \item \textbf{Rationale:} Fast iteration, no gas costs, easy debugging
            \item \textbf{Trade-off:} Mainnet deployment requires additional setup
            \item \textbf{Path Forward:} Prepared for deployment to production networks
        \end{itemize}
    \end{block}
\end{frame}
