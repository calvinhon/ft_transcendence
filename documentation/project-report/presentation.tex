\begin{frame}{Security Layers}
\begin{itemize}

\item Layer 5: Data Protection (Vault, Database Security)
\item Layer 6: Monitoring \& Logging
\item Layer 7: Incident Response
\end{itemize}

    mapping(uint256 => mapping(uint256 => uint256)) 
        public tournamentRankings;

        tournamentRankings[tournamentId][playerId] = rank;
    }

\begin{itemize}
