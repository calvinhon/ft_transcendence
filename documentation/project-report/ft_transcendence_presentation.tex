            \item \textbf{Real-time:} WebSocket protocol (Socket.io)
            \item \textbf{Blockchain:} Hardhat + Solidity smart contracts
            \item \textbf{API Gateway:} Nginx reverse proxy with TLS/WAF
        \end{itemize}
    \end{block}

% SLIDE 8: SECURITY IMPLEMENTATION
% ============================================================================
\begin{frame}{Coverage/Content: Security Implementation}
    \begin{block}{Defense-in-Depth (7 Layers)}
        \begin{enumerate}
            \small
            \item \textbf{Network:} HTTPS + TLS 1.2+, Nginx, ModSecurity WAF
            \item \textbf{Application:} Input validation, parameterized SQL, CSRF protection
            \item \textbf{Authentication:} JWT tokens (HS256), TOTP 2FA, RBAC
            \item \textbf{Data Protection:} bcrypt passwords, Vault secrets, encryption
            \item \textbf{Secrets Management:} HashiCorp Vault for credentials
            \item \textbf{WAF Rules:} ModSecurity blocks SQL injection, XSS, CSRF
            \item \textbf{Monitoring:} Security alerts, audit logging
        \end{enumerate}
    \end{block}
    
    \begin{block}{Security Measures}
        \begin{itemize}
            \small
            \item SQL Injection prevention: All queries parameterized with `?` placeholders
            \item XSS prevention: Content-Security-Policy headers
            \item CSRF protection: SameSite cookie attribute
            \item Password hashing: bcrypt with cost factor 10+
            \item Session handling: Secure and HttpOnly flags
        \end{itemize}
    \end{block}
\end{frame}

% SLIDE 17: LIMITATIONS OF APPLICATION

\begin{frame}{Limitations of the Application}
    \begin{block}{Current Limitations}
        \begin{itemize}
            \small
            \item \textbf{Single Game Type:} Only Pong implemented (by design scope)
            \item \textbf{Max 2 Players:} Game designed for 1v1 matches, not multiplayer FFA
            \item \textbf{Single Node Database:} SQLite cannot horizontally scale across multiple instances
            \item \textbf{Local Blockchain Only:} Smart contracts deployed on local Hardhat testnet, not Ethereum mainnet
            \item \textbf{Basic AI:} Single difficulty level for AI opponent in current version
            \item \textbf{Web-Only:} No native mobile apps (only responsive web design)
            \item \textbf{Limited Concurrency:} Single instance handles ~100 concurrent WebSocket connections
        \end{itemize}
    \end{block}
    
    \begin{block}{Architectural Limitations}
        \begin{itemize}
            \small
            \item No automatic service discovery (services hardcoded in configuration)
            \item Manual scaling required for horizontal expansion
            \item No persistent cache layer (would improve performance)
            \item Single database per service (shared database could enable joins)
        \end{itemize}
    \end{block}

\begin{frame}{Challenges \& Solutions}
    \begin{block}{Challenge 1: Real-Time Game Synchronization}
        \begin{itemize}
            \small
            \item \textbf{Problem:} Network latency causes gameplay desynchronization between clients

        A: Data export (JSON), deletion, consent tracking, retention policies. All implemented and verified by external audit.
